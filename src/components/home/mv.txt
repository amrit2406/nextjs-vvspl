"use client";
import { motion, useMotionValue } from "framer-motion";
import { useEffect, useRef } from "react";
import {

// --- Floating Geometric Orbs Background ---
const FloatingOrbs = () => {
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const mouseX = useMotionValue(-1000);
  const mouseY = useMotionValue(-1000);

  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    const ctx = canvas.getContext("2d");
    if (!ctx) return;

    let animationFrameId: number;
    let orbs: Orb[] = [];
    const orbCount = 12;

    const resize = () => {
      canvas.width = window.innerWidth;
      canvas.height = window.innerHeight;
    };

    window.addEventListener("resize", resize);
    resize();

    class Orb {
      x: number; y: number; baseX: number; baseY: number;
      vx: number; vy: number; radius: number; rotation: number;
      color: string; glowColor: string; rotationSpeed: number;

      constructor() {
        this.x = Math.random() * canvas!.width;
        this.y = Math.random() * canvas!.height;
        this.baseX = this.x;
        this.baseY = this.y;
        this.vx = (Math.random() - 0.5) * 0.8;
        this.vy = (Math.random() - 0.5) * 0.8;
        this.radius = 40 + Math.random() * 60;
        this.rotation = Math.random() * Math.PI * 2;
        this.rotationSpeed = (Math.random() - 0.5) * 0.02;
        
        const colors = [
          { base: '#FF7E00', glow: '#FFA726' },
          { base: '#044DB6', glow: '#42A5F5' },
          { base: '#FFB74D', glow: '#FFCC80' },
          { base: '#26C6DA', glow: '#4DD0E1' }
        ];
        const color = colors[Math.floor(Math.random() * colors.length)];
        this.color = color.base;
        this.glowColor = color.glow;
      }

      update() {
        this.baseX += this.vx;
        this.baseY += this.vy;
        if (this.baseX > canvas!.width + this.radius || this.baseX < -this.radius) this.vx *= -1;
        if (this.baseY > canvas!.height + this.radius || this.baseY < -this.radius) this.vy *= -1;

        // Mouse repulsion
        const dx = mouseX.get() - this.x;
        const dy = mouseY.get() - this.y;
        const distance = Math.sqrt(dx * dx + dy * dy);
        if (distance < 250) {
          this.x = this.baseX - (dx * 0.08);
          this.y = this.baseY - (dy * 0.08);
        } else {
          this.x = this.baseX;
          this.y = this.baseY;
        }

        this.rotation += this.rotationSpeed;
      }

      draw() {
        if (!ctx) return;
        
        // Glow effect
        const gradient = ctx.createRadialGradient(
          this.x, this.y, 0,
          this.x, this.y, this.radius * 1.5
        );
        gradient.addColorStop(0, `${this.glowColor}80`);
        gradient.addColorStop(0.3, `${this.glowColor}40`);
        gradient.addColorStop(1, 'transparent');
        
        ctx.save();
        ctx.translate(this.x, this.y);
        ctx.rotate(this.rotation);
        
        // Outer glow
        ctx.fillStyle = gradient;
        ctx.beginPath();
        ctx.arc(0, 0, this.radius * 1.5, 0, Math.PI * 2);
        ctx.fill();
        
        // Inner orb with geometric pattern
        ctx.fillStyle = this.color;
        ctx.shadowColor = this.glowColor;
        ctx.shadowBlur = 20;
        
        ctx.beginPath();
        ctx.arc(0, 0, this.radius * 0.7, 0, Math.PI * 2);
        ctx.fill();
        
        // Geometric cutouts
        ctx.fillStyle = 'rgba(255, 255, 255, 0.2)';
        for (let i = 0; i < 6; i++) {
          ctx.save();
          ctx.rotate((Math.PI * 2 / 6) * i);
          ctx.beginPath();
          ctx.arc(0, 0, this.radius * 0.3, 0, Math.PI * 2);
          ctx.fill();
          ctx.restore();
        }
        
        ctx.shadowBlur = 0;
        ctx.restore();
      }
    }

    const init = () => {
      orbs = [];
      for (let i = 0; i < orbCount; i++) orbs.push(new Orb());
    };

    const animate = () => {
      ctx.fillStyle = 'rgba(255, 255, 255, 0.1)';
      ctx.fillRect(0, 0, canvas.width, canvas.height);
      
      orbs.forEach(orb => { 
        orb.update(); 
        orb.draw(); 
      });
      
      animationFrameId = requestAnimationFrame(animate);
    };

    init();
    animate();

    const handleMouseMove = (e: MouseEvent) => {
      mouseX.set(e.clientX);
      mouseY.set(e.clientY);
    };

    window.addEventListener("mousemove", handleMouseMove);
    return () => {
      window.removeEventListener("resize", resize);
      window.removeEventListener("mousemove", handleMouseMove);
      cancelAnimationFrame(animationFrameId);
    };
  }, [mouseX, mouseY]);

  return <canvas ref={canvasRef} className="absolute inset-0 z-0 pointer-events-none" />;
};
